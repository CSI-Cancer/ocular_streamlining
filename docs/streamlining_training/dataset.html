 44def tube_train_test_split(slides,
 45                           random_state=43,
 46                             test_size=0.2):
 47    # Train val split the tubes
 48    tubes = np.unique([slide[0:-2] for slide in slides])
 49    train_tubes, val_tubes = train_test_split(tubes,
 50                                              test_size=test_size,
 51                                              random_state=random_state)
 52    train_slides = [slide for slide in slides if slide[0:-2] in train_tubes]
 53    val_slides = [slide for slide in slides if slide[0:-2] in val_tubes]
 54    return train_slides, val_slides

 57def get_events_and_perML(
 58        slide_ids,

 62    ## Get per ML data from the database

194def get_data_from_files(slide_id: str,
195                         ocular_params: dict,
196                         ocular_path: Path = PRODUCTION_REPORT_DIR):  
197    file_holder = {}
198    if ocular_params["dapipos"]:
199        event_files = ocular_params["dapi_pos_files"]
200    else:
201        event_files = ocular_params["dapi_neg_files"]
202    
203    failed_loads = []
204    empty_files = []
205
206    for file in event_files:
207        file_path = ocular_path / slide_id / "ocular" / file
208
209        try:
210            file_holder[file] = pyreadr.read_r(file_path)
211        except Exception as e:
212            logger.error(f"Traceback while loading files for the slide : {slide_id} \n {e}")
213            failed_loads.append(file)
214            continue
215        file_holder[file] = file_holder[file][None]
216        if (file == "ocular_interesting.rds" or file == "ocular_not_interesting.rds"
217            ) and ocular_params["drop_common_cells"]:
218            file_holder[file] = file_holder[file][
219                file_holder[file]["catalogue_classification"] != "common_cell"]
220        if len(file_holder[file]) == 0:
221            empty_files.append(file)
222        else:
223            if ocular_params["dapipos"]:
224                file_holder[file]["rowname"] = file_holder[file]["rowname"].astype(str)
225                split_res = file_holder[file]["rowname"].str.split(" ", n=1, expand=True)
226
227                if len(split_res.columns) != 2:
228                        logger.error(f"Slide {slide_id} rowname column has issues")
229                file_holder[file][["frame_id", "cell_id"]] = split_res
230                file_holder[file][
231                        ["frame_id", "cell_id"]
232                ] = file_holder[file][["frame_id", "cell_id"]].astype(
233                    "int"
234                )
235            file_holder[file].reset_index(drop=True, inplace=True)
236
237    for empty in empty_files:
238        logger.info(f"deleting empty file: {empty} for the slide: {slide_id}")
239        del file_holder[empty]
240    
241    for dailed in failed_loads:
242        logger.error(f"Deleting failed to load file: {failed_loads} for the slide: {slide_id}")
243        del file_holder[failed_loads]
244    
245    ocular_data = [
246            file_holder[filename]
247            for filename in file_holder.keys()
248        ]
249    if len(ocular_data) != 0:
250        ocular_data = pd.concat(ocular_data)
251        logger.info(f"Found {len(ocular_data)} events for slide {slide_id} in ocular files")
252
253    return ocular_data

255def get_frame_morphs(slide_id, ocular_params,
256                     path: Path = PRODUCTION_REPORT_DIR):
257
258    slide__path = path / slide_id / "ocular" 
259    morph_files = ["framestat-means.csv", "framestat-dev.csv"]

265        df = df.rename(columns={df.columns[0]: 'frame_id'})

267        files[prefix] = df

285    slide_mean = slide_mean.add_prefix("slide_means_")\

290def calculate_sdom(df, feature, prefix):
291    feature_mean = prefix + "_means_" + feature
292    feature_sd = prefix + "_dev_" + feature
293    df[prefix+"_sdom_"+feature] = df[feature] - df[feature_mean]
294    df[prefix+"_sdom_"+feature] = df[prefix+"_sdom_"+feature].divide(df[feature_sd])
295    df[prefix+"_sdom_"+feature] = df[prefix+"_sdom_"+feature].replace([np.inf, -np.inf], 0)
296    if df[prefix+"_sdom_"+feature].isna().any():
297        n_of_nas = df[prefix+"_sdom_"+feature][df[prefix+"_sdom_"+feature].isna()].index
298        logger.info(
299            f"Replacing nan with zero for {prefix}_sdom_{feature} , number of entries: {len(n_of_nas)}"
300        )
301        df[prefix+"_sdom_"+feature].fillna(0, inplace=True)
302    return df[prefix+"_sdom_"+feature]

349def process_slide(slide_id,
350                     events,
351                       per_ml,
352                        ocular_params,):
353    """
354    This method accepts the slide id, curated events and per ml data,
355    drops the duplicates and drops unused columns, and gets the frame and slide level
356    morphopmetrics and merges them along with event level data then reads ocular files,
357    namely, rc-final1.rds, rc-final2.rds, rc-final3.rds, rc-final4.rds, to get the set
358    of interesting events along with manually curated interesting events and then set
359    of not interesting events from uncurated_not_interesting evenst from files and,
360    manual_not_interesting to produce the final dataframe for that slide.
361
362    Args:
363    slide_id: str
364    events: pd.DataFrame
365    per_ml: pd.DataFrame
366    ocular_params: dict
367
368    Returns:
369    final_df: pd.DataFrame
370
371    """
372    # Slice events corresponding to the slide
373    manual_interesting = events[(events['slide_id'] == slide_id) & (events['interesting'] == 1)].copy()
374    manual_not_interesting = events[(events['slide_id'] == slide_id) & (events['interesting'] == 0)].copy()
375    manual_interesting["unique_id"] = (
376        manual_interesting["slide_id"].astype(str) + "_" +
377        manual_interesting["frame_id"].astype(str) + "_" +
378        manual_interesting["cell_id"].astype(str)
379        )
380    manual_not_interesting["unique_id"] = (
381        manual_not_interesting["slide_id"].astype(str) + "_" +
382        manual_not_interesting["frame_id"].astype(str) + "_" +
383        manual_not_interesting["cell_id"].astype(str)
384        )
385    top_data = get_top_level_data(slide_id, per_ml)
386
387    uncurated_events = get_data_from_files(slide_id, ocular_params)
388    if len(uncurated_events) != 0:
389        uncurated_events["slide_id"] = slide_id
390
391    uncurated_events["unique_id"] = (
392        uncurated_events["slide_id"].astype(str) + "_" +
393        uncurated_events["frame_id"].astype(str) + "_" +
394        uncurated_events["cellx"].astype(str) + "_" +
395        uncurated_events["celly"].astype(str)
396    )
397    uncurated_events.sort_values(by=["cell_id"],
398                                  ascending=True, inplace=True)
399    uncurated_events.drop_duplicates(subset=['unique_id'],
400                                      keep='first', inplace=True)
401    uncurated_events["unique_id"] = (
402        uncurated_events["slide_id"].astype(str) + "_" +
403        uncurated_events["frame_id"].astype(str) + "_" +
404        uncurated_events["cell_id"].astype(str)
405    )
406    uncurated_events.drop_duplicates(subset=['unique_id'],
407                                      keep='first', inplace=True)
408    
409    frame_stats = get_frame_morphs(slide_id, ocular_params)
410
411    slide_stats = get_slide_stats(slide_id, ocular_params)
412
413    uncurated_events = merge_all(uncurated_events,
414                         frame_stats,
415                         slide_stats)
416    if ocular_params['drop_columns']:
417        columns_to_keep = uncurated_events.columns
418        if 'haralick' in ocular_params['drop_morphs']:
419            columns_to_keep = [col for col in columns_to_keep if '.h.' not in col]
420        if 'theta' in ocular_params['drop_morphs']:
421            columns_to_keep = [col for col in columns_to_keep if 'theta' not in col]
422            
423        if 'blurred' in ocular_params['drop_morphs']:
424            for channel in channel_map.keys():
425                columns_to_keep = [col for col in columns_to_keep if 'B'+channel_map[channel] not in col]
426
427        if 'multi_channel' in ocular_params['drop_morphs']:
428            drop_keys = [''.join(pair) for pair in permutations(channel_map.values(), 2)]
429            for key in drop_keys:
430                columns_to_keep = [col for col in columns_to_keep if key not in col]
431
432        if 'mean_sd_q05' in ocular_params['drop_morphs']:
433            columns_to_keep=[col for col in columns_to_keep if ('.mean' in col) or ('.sd' in col) or ('.q05' in col) ]
434        
435        if 'mean_q05' in ocular_params['drop_morphs']:
436            columns_to_keep=[col for col in columns_to_keep if ('.mean' in col) or ('.q05' in col) ]
437        
438        if 'remove_slide_level' in ocular_params['drop_morphs']:
439            columns_to_keep=[col for col in columns_to_keep if 'slide' not in col ]
440        
441        if 'remove_frame_level' in ocular_params ['drop_morphs']:
442            columns_to_keep=[col for col in columns_to_keep if 'frame' not in col ]
443
444        # Adding the frame and slide id since it was dropped during morphs drop
445        columns_to_keep+=["frame_id", "slide_id"]
446        compute_features = []
447        for prefix in feature_sdom_levels.keys():
448            compute_features += [prefix +"_sdom_"+item for item in feature_sdom_levels[prefix]]
449        columns_to_keep += compute_features
450
451        uncurated_events = generate_sdom(uncurated_events, feature_sdom_levels, compute_features)
452
453
454        uncurated_events = uncurated_events[columns_to_keep]
455
456        uncurated_interesting = uncurated_events[(
457        uncurated_events['unique_id'].isin(manual_interesting['unique_id']) )].copy()
458
459        uncurated_not_interesting = uncurated_events[(
460        ~(uncurated_events['unique_id'].isin(manual_interesting['unique_id'])) )]
461
462        manual_not_interesting = uncurated_not_interesting[uncurated_not_interesting['unique_id'].isin(
463                                                            manual_not_interesting['unique_id'])]
464        
465        uncurated_not_interesting = uncurated_not_interesting[~(uncurated_not_interesting['unique_id'].isin(
466                                                            manual_not_interesting['unique_id']))]
467
468        if ocular_params['not_int_sample_size'] > 0:
469            if len(manual_not_interesting) > ocular_params['not_int_sample_size']:
470                manual_not_interesting=manual_not_interesting.sample(n=ocular_params['not_int_sample_size'],
471                                                                 random_state=42)
472        uncurated_interesting["interesting"] = 1
473        uncurated_not_interesting["interesting"] = 0
474        manual_not_interesting["interesting"] = 0
475
476        final_df = pd.concat([uncurated_interesting, uncurated_not_interesting,
477                              manual_not_interesting])
478        return final_df
479

481def get_slides_data(slide_ids: list,
482                    events: pd.DataFrame,
483                    per_ml: pd.DataFrame,
484                    ocular_params: dict = ocular_params,
485                    ):
486    slide_data = []
487    for slide_id in slide_ids:
